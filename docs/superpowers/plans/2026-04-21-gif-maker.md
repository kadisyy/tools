# GIF Maker Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 纯前端图片转 GIF 工具，上传一张图片，选择动效（抖动/闪烁/渐入渐出/缩放脉冲），调节速度，生成并下载动态 GIF。

**Architecture:** 单页 `gif-maker.html`，分屏布局（与 meme-maker 一致）。左侧控制面板，右侧 Canvas 实时预览 + 生成/下载区。使用 gif.js（CDN）在 Web Worker 中编码 GIF，Canvas 2D API 逐帧渲染动效。首页新增 TOOL_004 卡片。

**Tech Stack:** 纯 HTML5 + CSS3 + Vanilla JS，gif.js（CDN），Canvas 2D API，FileReader API。

```js
// ── DOM refs ──
const fileInput    = document.getElementById('fileInput');
const uploadZone   = document.getElementById('uploadZone');
const imgPreview   = document.getElementById('imgPreview');
const genBtn       = document.getElementById('genBtn');
const emptyState   = document.getElementById('emptyState');
const previewWrap  = document.getElementById('previewWrap');
const previewCanvas= document.getElementById('previewCanvas');
const progressWrap = document.getElementById('progressWrap');
const progressLabel= document.getElementById('progressLabel');
const progressFill = document.getElementById('progressFill');
const dlBtn        = document.getElementById('dlBtn');

// ── State ──
let sourceImage  = null;
let selectedEffect = 'shake';
let selectedSpeed  = 'medium';
let previewRafId   = null;
let previewFrame   = 0;

// ── Speed config (ms per frame, total frames) ──
const SPEED_CONFIG = {
  slow:   { delay: 80,  frames: 20 },
  medium: { delay: 50,  frames: 16 },
  fast:   { delay: 30,  frames: 12 },
};

// ── Upload ──
function loadImage(file) {
  if (!file || !file.type.startsWith('image/')) return;
  const reader = new FileReader();
  reader.onload = e => {
    const img = new Image();
    img.onload = () => {
      sourceImage = img;
      imgPreview.src = e.target.result;
      imgPreview.style.display = 'block';
      uploadZone.querySelector('.upload-icon').style.display = 'none';
      uploadZone.querySelector('.upload-hint').style.display = 'none';
      emptyState.style.display = 'none';
      previewWrap.style.display = 'block';
      genBtn.disabled = false;
      dlBtn.style.display = 'none';
      startPreview();
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

fileInput.addEventListener('change', e => loadImage(e.target.files[0]));
uploadZone.addEventListener('click', () => fileInput.click());
uploadZone.addEventListener('dragover', e => { e.preventDefault(); uploadZone.classList.add('drag-over'); });
uploadZone.addEventListener('dragleave', () => uploadZone.classList.remove('drag-over'));
uploadZone.addEventListener('drop', e => {
  e.preventDefault();
  uploadZone.classList.remove('drag-over');
  loadImage(e.dataTransfer.files[0]);
});

// ── Effect selection ──
document.querySelectorAll('.effect-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.effect-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    selectedEffect = btn.dataset.effect;
    if (sourceImage) startPreview();
  });
});

// ── Speed selection ──
document.querySelectorAll('.speed-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.speed-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    selectedSpeed = btn.dataset.speed;
    if (sourceImage) startPreview();
  });
});

// ── Canvas preview (RAF loop) ──
const SIZE = 480;

function startPreview() {
  if (previewRafId) cancelAnimationFrame(previewRafId);
  previewCanvas.width = SIZE;
  previewCanvas.height = SIZE;
  previewFrame = 0;
  const { frames, delay } = SPEED_CONFIG[selectedSpeed];
  let lastTime = 0;

  function loop(ts) {
    if (ts - lastTime >= delay) {
      renderFrame(previewCanvas, sourceImage, selectedEffect, previewFrame, frames);
      previewFrame = (previewFrame + 1) % frames;
      lastTime = ts;
    }
    previewRafId = requestAnimationFrame(loop);
  }
  previewRafId = requestAnimationFrame(loop);
}

// ── Frame renderer ──
function renderFrame(canvas, img, effect, frame, totalFrames) {
  const ctx = canvas.getContext('2d');
  const t = frame / totalFrames; // 0..1 progress

  ctx.clearRect(0, 0, SIZE, SIZE);
  ctx.save();

  // Cover-fit draw helper
  function drawImg(dx, dy, dw, dh) {
    const scale = Math.max(dw / img.width, dh / img.height);
    const sw = img.width * scale, sh = img.height * scale;
    const sx = dx + (dw - sw) / 2, sy = dy + (dh - sh) / 2;
    ctx.drawImage(img, sx, sy, sw, sh);
  }

  if (effect === 'shake') {
    const amp = 8;
    const dx = Math.round(Math.sin(t * Math.PI * 4) * amp);
    const dy = Math.round(Math.cos(t * Math.PI * 3) * amp * 0.6);
    ctx.translate(dx, dy);
    drawImg(0, 0, SIZE, SIZE);

  } else if (effect === 'blink') {
    const visible = frame % 4 < 2; // blink every 2 frames
    if (visible) drawImg(0, 0, SIZE, SIZE);
    else {
      drawImg(0, 0, SIZE, SIZE);
      ctx.fillStyle = 'rgba(255,255,255,0.85)';
      ctx.fillRect(0, 0, SIZE, SIZE);
    }

  } else if (effect === 'fade') {
    // 0→1→0 alpha cycle
    const alpha = t < 0.5 ? t * 2 : (1 - t) * 2;
    ctx.globalAlpha = Math.max(0.05, alpha);
    drawImg(0, 0, SIZE, SIZE);

  } else if (effect === 'zoom') {
    // scale 1.0 → 1.12 → 1.0
    const scale = 1 + Math.sin(t * Math.PI) * 0.12;
    ctx.translate(SIZE / 2, SIZE / 2);
    ctx.scale(scale, scale);
    ctx.translate(-SIZE / 2, -SIZE / 2);
    drawImg(0, 0, SIZE, SIZE);
  }

  ctx.restore();
}

function generateGif() {
  if (!sourceImage) return;
  const { delay, frames } = SPEED_CONFIG[selectedSpeed];

  // UI: show progress, hide download
  genBtn.disabled = true;
  progressWrap.style.display = 'flex';
  progressFill.style.width = '0%';
  progressLabel.textContent = 'GENERATING... 0%';
  dlBtn.style.display = 'none';

  // Off-screen canvas for frame rendering
  const offCanvas = document.createElement('canvas');
  offCanvas.width = SIZE;
  offCanvas.height = SIZE;

  const gif = new GIF({
    workers: 2,
    quality: 10,
    width: SIZE,
    height: SIZE,
    workerScript: 'https://cdn.jsdelivr.net/npm/gif.js@0.2.0/dist/gif.worker.js',
  });

  // Add frames
  for (let i = 0; i < frames; i++) {
    renderFrame(offCanvas, sourceImage, selectedEffect, i, frames);
    gif.addFrame(offCanvas, { copy: true, delay });
  }

  gif.on('progress', p => {
    const pct = Math.round(p * 100);
    progressFill.style.width = pct + '%';
    progressLabel.textContent = `GENERATING... ${pct}%`;
  });

  gif.on('finished', blob => {
    progressWrap.style.display = 'none';
    genBtn.disabled = false;

    const url = URL.createObjectURL(blob);
    dlBtn.style.display = 'inline-block';
    dlBtn.onclick = () => {
      const a = document.createElement('a');
      a.href = url;
      a.download = `gif-maker-${selectedEffect}-${selectedSpeed}.gif`;
      a.click();
    };
  });

  gif.render();
}
```
